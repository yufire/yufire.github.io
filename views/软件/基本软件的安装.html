目前只是 Centos和Ubuntu的安装教程

# 一、安装 JDK

# Centos--Ubuntu通用安装

  • 复制 jdk 的安装包到你想要的目录里 一般为 tar.gz
tar -zxvf jdk1.8.0_171.tar.gz  # 解压
1
  • 配置环境变量
vim /etc/profile  # 编辑环境变量
1
  • 往最后一行添加以下内容
export JAVA_HOME=/usr/lib/jdk/jdk1.8.0_171  # 你的jdk路径
export JRE_HOME=${JAVA_HOME}/jre    
export CLASSPATH=.:${JAVA_HOME}/lib:${JRE_HOME}/lib    
export PATH=${JAVA_HOME}/bin:$PATH
1
2
3
4
  • wq! 保存退出
source /etc/profile # 重启配置文件
java -version
1
2

# Centos在线安装

yum update  # 更新yum源
yum search jdk  # 找到你想安装的版本
yun install java-1.8.0-openjdk.x86_64 -y  # 安装  -y 为自动同意xxx
java -version

# 二、安装Tomcat

  • 复制 Tomcat 安装包到你想要的目录
tar -zxvf tomcat8.tar.gz  # 解压缩
1
  • 启动 Tomcat
cd /usr/local/apps/tomcat/bin  # 进入 tomcat的bin目录
./startup.sh   # 启动
./shutdown.sh  # 停止
1
2
3

# 小坑

  • 如果你的项目有生成验证码的代码的话 需要在 catalina.sh 里添加以下代码
JAVA_OPTS="$JAVA_OPTS -Djava.awt.headless=true"
1

# 三、安装MySQL

# Centos安装 tar.gz

  • 将MySQL的压缩文件上传到你想要的目录
tar -zxvf mysql5.6_an_zhuang_bao.tar.gz  # 解压
1
  • 添加 mysql 的用户组和mysql用户
groupadd mysql  # 创建mysql用户组
useradd -g mysql mysql  # 添加mysql用户
cd /usr/local/apps/mysql  # 切换到你的mysql目录
chown -R mysql:mysql ./  # 赋予mysql用户拥护执行该文件夹中的文件的权限
1
2
3
4
  • 在 /etc 中创建 my.cnf MySQL的配置文件
vim /etc/my.cnf
1
  • 将以下内容复制
[mysql]
# 设置mysql客户端默认字符集
default-character-set=utf8
socket=/var/lib/mysql/mysql.sock

[mysqld]
skip-name-resolve
#设置3306端口
port = 3306
socket=/var/lib/mysql/mysql.sock
# 设置mysql的安装目录  你的MySQL目录!!!
basedir=/usr/local/apps/mysql5.6
# 设置mysql数据库的数据的存放目录  你的MySQL目录!!!
datadir=/usr/local/apps/mysql5.6/data
# 允许最大连接数
max_connections=200
# 服务端使用的字符集默认为8比特编码的latin1字符集
character-set-server=utf8
# 创建新表时将使用的默认存储引擎
default-storage-engine=INNODB
lower_case_table_name=1
max_allowed_packet=16M

# Ubuntu安装MySQL

安装mysql

# MySQL初始化时的统一设置

mysql -uroot -p # 两下回车 第一次登录没有密码
use mysql;  # 选中mysql数据库
update user set password=password('root') where user='root'; # 修改root的密码
grant all on *.* to root@'%' identified by '你的密码' with grant option; # 添加远程访问的权限
flush privileges;    # 刷新权限
exit; # 退出

service mysql restart # 重启MySQL服务
