  • Swagger是什么？它能干什么？

  • 提供描述、生产、消费和可视化RESTful Web Service。

  1. 在项目文价夹中创建config目录 config目录下创建 SwaggerConfig配置类

  2. SwaggerConfig配置类 修改

package cn.yufire.config;

import com.google.common.collect.Lists;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import springfox.documentation.builders.PathSelectors;
import springfox.documentation.builders.RequestHandlerSelectors;
import springfox.documentation.service.ApiInfo;
import springfox.documentation.service.Contact;
import springfox.documentation.spi.DocumentationType;
import springfox.documentation.spring.web.plugins.Docket;
import springfox.documentation.swagger2.annotations.EnableSwagger2;

import java.util.ArrayList;

@Configuration    //标注这是个配置类
@EnableSwagger2  //开启Swagger2
public class SwaggerConfig {


    //多人协作开发可以扫描自己的包

    /**
     * 分组1
     * @return
     */
    @Bean
    public Docket docket1() {
        return new Docket(DocumentationType.SWAGGER_2)
                .apiInfo(apiInfo())
                .groupName("1")
                .select()
                .apis(RequestHandlerSelectors.basePackage("cn.yufire.controller"))
              //.paths(PathSelectors.any())
                .build();
    }
    
    /**
     * 分组2  分组n.....
     * @return
     */
    @Bean
    public Docket docket2() {
        return new Docket(DocumentationType.SWAGGER_2)
                .apiInfo(apiInfo())
                .groupName("1")
                .select()
                .apis(RequestHandlerSelectors.basePackage("cn.yufire.controller"))
              //.paths(PathSelectors.any())
                .build();
    }
    
    
   
    /**
    * 配置 项目版本号 作者信息 等等的类
    */
    private ApiInfo apiInfo() {
        //作者信息
        Contact contact = new Contact("yufire", "http://localhost:8080", "2867874665@qq.com");
        return new ApiInfo(
                "yufire的API文档",
                "yufire的API文档",
                "1.0",
                "https://www.yuweb.cf", //作者说明网址
                contact, //作者信息
                "Apache 2.0", //开源协议
                "http://www.apache.org/licenses/LICENSE-2.0", //开源协议网址
                new ArrayList());
    }

}

  1. Swagger的常用注解说明

# 这些注解的作用只是在Swagger生成的接口文档上添加中文解释

# @Api() 该Api 用在类上，说明该类的作用。可以标记一个Controller类做为swagger 文档资源，使用方式：
@Api(value = "/user", description = "UserController这是User的控制类")
1

# @ApiOperation() ：用在方法上，说明方法的作用，每一个url资源的定义,使用方式：
@ApiOperation( "请求新闻数据的接口") //标注这个接口
1

# @ApiParam() :请求属性 这个接口需要什么数据,使用方式：
public String getParams(@RequestParam("name")String name,@RequestParam("age")Integer age){

}
